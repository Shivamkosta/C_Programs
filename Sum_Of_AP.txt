#include<stdio.h>

void main()
{
   int a,n,d,tn,sum=0;
   printf("Enter first number of the A.P series :");
   scanf("%d",&a);
   
   printf("Enter total number of A.P series is : ");
   scanf("%d",&n);
   
   printf("Enter total number of difference :");
   scanf("%d",&d);
   
   //formula find sum of AP
   sum = (n*(2*a + (n-1)*d))/2;
   tn = (a + (n-1)*d);
   
   for(int i = a; i<=tn; i=i+d)
   {
       if(i!=tn)
       printf("%d +",i);
       else
       printf("%d = %d",i,sum);
   }
    
    getch();
}

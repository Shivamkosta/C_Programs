#include <stdio.h>  
int main(){
      int n,r,ncr;
      printf("Enter an integer to check ncr factor :");
      scanf("%d %d \n",&n,&r);
      ncr = fact(n)/(fact(r)*fact(n-r));
      printf("ncr factor of %d and %d is %d ",n,r,ncr);
      return 0;
}
int fact(int n){
    int i = 1;
    while(n!=0){
        i = i*n;
        n--;
    }
    return i;
}

#include <stdio.h>

int main()
{
    float number,sum=0,i;
	
    printf("Enter the number :");
    scanf("%f",&number);
    
    for(i = 1; i<=number; i++)
    {
        sum = sum + (1/i);
        
        if(i == 1)
        {
            printf("\n1+");
        }
        else if(i == number)
        {
            printf("(1/%2f)",i);
        }
        else
        {
            printf("(1/%2f)+",i);
        }
    }
    printf("\n\n The sum of Series is :%.2f\n",sum);
    return 0;
}
